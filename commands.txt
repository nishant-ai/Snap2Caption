### Merge Models

git clone -b model_serving_endpoint https://github.com/nishant-ai/Snap2Caption.git

### Docker File for Model Optimization
docker build -f Dockerfile.optimization -t merge-model .

docker run --rm \
  --gpus all   \
  -v $(pwd)/config_model_inference.yaml:/app/config_model_inference.yaml \
  -v $(pwd)/model_optimization.py:/app/model_optimization.py \
  -v $(pwd)/models:/models \
  merge-model


### Build Docker FIle for Fast API 

docker build -f Dockerfile.fastapi -t llava-fastapi-server .

docker run --gpus all --rm \
  -p 8000:8000 \
  -v /home/cc/Snap2Caption/models/LLAVA_MERGED:/lora_adapters \
  -v /home/cc/Snap2Caption:/app \
  llava-fastapi-server
